let accounts = JSON.parse(localStorage.getItem('payit_accounts')) || [{ id: '1', name: 'Privé' }, { id: '2', name: 'VOF' }]; let transactions = JSON.parse(localStorage.getItem('payit_transactions')) || []; let activeAccountId = accounts[0].id; let editingId = null; let sortConfig = { field: 'status', direction: 'asc' }; function renderAccounts() { const list = document.getElementById('accounts-list'); if(!list) return; list.innerHTML = ''; const displayAccounts = [...accounts].sort((a, b) => a.name.localeCompare(b.name)); displayAccounts.forEach(acc => { const item = document.createElement('div'); item.className = 'account-item'; item.innerHTML = `
${acc.name}
`; list.appendChild(item); }); const activeAcc = accounts.find(a => a.id === activeAccountId) || accounts[0]; document.getElementById('current-account-title').innerText = activeAcc.name; updateAccountDropdown(); } function updateAccountDropdown() { const select = document.getElementById('target-account'); select.innerHTML = ''; accounts.sort((a, b) => a.name.localeCompare(b.name)).forEach(acc => { const opt = document.createElement('option'); opt.value = acc.id; opt.innerText = acc.name; select.appendChild(opt); }); } function selectAccount(id) { activeAccountId = id; renderAccounts(); renderTransactions(); } function renderTransactions() { const tbody = document.getElementById('transactions'); if(!tbody) return; tbody.innerHTML = ''; let data = transactions.filter(t => (t.accountId || accounts[0].id) === activeAccountId); data.sort((a, b) => { let vA = a[sortConfig.field] || ""; let vB = b[sortConfig.field] || ""; if (sortConfig.field === 'amount') { vA = parseFloat(a.amount) || 0; vB = parseFloat(b.amount) || 0; } if (sortConfig.field === 'status') { vA = a.status === 'pending' ? 0 : 1; vB = b.status === 'pending' ? 0 : 1; } if (vA < vB) return sortConfig.direction === 'asc' ? -1 : 1; if (vA > vB) return sortConfig.direction === 'asc' ? 1 : -1; return (b.date || "").localeCompare(a.date || ""); }); data.forEach(t => { const tr = document.createElement('tr'); tr.onclick = () => openModal(t.id); tr.innerHTML = ` ${t.date || ''} ${t.category || ''} € ${parseFloat(t.amount || 0).toFixed(2)} ${t.memo || ''} `; tbody.appendChild(tr); }); } function toggleSort(field) { sortConfig.direction = (sortConfig.field === field && sortConfig.direction === 'asc') ? 'desc' : 'asc'; sortConfig.field = field; renderTransactions(); } function openModal(id = null) { editingId = id; const m = document.getElementById('modal'); const accSelect = document.getElementById('account-select-container'); const editActions = document.getElementById('edit-actions'); if (id) { const t = transactions.find(x => x.id === id); document.getElementById('modal-title').innerText = 'Bewerken'; document.getElementById('date').value = t.date; document.getElementById('category').value = t.category; document.getElementById('memo').value = t.memo || ''; document.getElementById('amount').value = t.amount; document.getElementById('status').value = t.status; document.getElementById('target-account').value = t.accountId || accounts[0].id; accSelect.style.display = 'block'; editActions.style.display = 'flex'; } else { document.getElementById('modal-title').innerText = 'Nieuw'; document.getElementById('date').value = new Date().toISOString().split('T')[0]; document.getElementById('category').value = ''; document.getElementById('memo').value = ''; document.getElementById('amount').value = ''; document.getElementById('status').value = 'pending'; accSelect.style.display = 'none'; editActions.style.display = 'none'; } m.style.display = 'flex'; } function closeModal() { document.getElementById('modal').style.display = 'none'; } function saveTransaction() { const data = { date: document.getElementById('date').value, category: document.getElementById('category').value || 'Diversen', memo: document.getElementById('memo').value, amount: document.getElementById('amount').value || 0, status: document.getElementById('status').value, accountId: editingId ? document.getElementById('target-account').value : activeAccountId }; if (editingId) { const i = transactions.findIndex(t => t.id === editingId); transactions[i] = { ...data, id: editingId }; } else { data.id = Date.now().toString(); transactions.push(data); } localStorage.setItem('payit_transactions', JSON.stringify(transactions)); closeModal(); renderTransactions(); } function duplicateTransaction() { if (!editingId) return; const t = transactions.find(x => x.id === editingId); const copy = { ...t, id: Date.now().toString(), accountId: activeAccountId }; transactions.push(copy); localStorage.setItem('payit_transactions', JSON.stringify(transactions)); closeModal(); renderTransactions(); } function addAccount() { const n = prompt("Naam rekening:"); if (n && n.trim()) { const newId = Date.now().toString(); accounts.push({ id: newId, name: n.trim() }); localStorage.setItem('payit_accounts', JSON.stringify(accounts)); activeAccountId = newId; renderAccounts(); renderTransactions(); } } function renameAccount(id) { const acc = accounts.find(a => a.id === id); const n = prompt("Nieuwe naam:", acc.name); if (n && n.trim()) { acc.name = n.trim(); localStorage.setItem('payit_accounts', JSON.stringify(accounts)); renderAccounts(); } } function removeAccount(id) { if (accounts.length <= 1) return; if (!confirm("Rekening verwijderen?")) return; if (!confirm("Weet je het zeker? Alle transacties worden dan ook verwijderd.")) return; accounts = accounts.filter(a => a.id !== id); transactions = transactions.filter(t => t.accountId !== id); activeAccountId = accounts[0].id; localStorage.setItem('payit_accounts', JSON.stringify(accounts)); localStorage.setItem('payit_transactions', JSON.stringify(transactions)); renderAccounts(); renderTransactions(); } function deleteTransaction() { if (editingId && confirm('Wissen?')) { transactions = transactions.filter(t => t.id !== editingId); localStorage.setItem('payit_transactions', JSON.stringify(transactions)); closeModal(); renderTransactions(); } } renderAccounts(); renderTransactions();
